  <!-- Обработка "radio"-->
  <script>
  function check_rad(val) {
   var inp = document.getElementsByName(val);
   for (var i = 0; i < inp.length; i++) {
    if (inp[i].type == "radio" && inp[i].checked) {
    return inp[i].value;
   }
  }
 }
  var radio_val = check_rad("flash") // radio_val = "on"
  </script>

   function check_sel(val) {
    var s = document.getElementById(val);
    for (var i = 0; i < s.options.length; i++) {
     if (s.options[i].selected) {
     return s.options[i].value
    }
   }
  }
  var select_val = check_sel("wifi_mode") // select_val = "st"

  </script>

   var close_m = document.getElementById("close_m");

   btn.onclick = function() {
    modal.style.opacity = "1";
    modal.style.display = "block";
   }
   close_m.onclick = function() {
    modal.style.opacity = "0";
    setTimeout(function() {
      modal.style.display = "none";
    }, 600);
   }
  </script>
